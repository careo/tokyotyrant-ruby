    def get(pkey)
      pkey = _argstr(pkey)
      args = Array.new
      args.push(pkey)
      rv = misc("get", args)
      return nil if !rv
      cols = Hash.new()
      cnum = rv.length
      cnum -= 1
      i = 0
      while i < cnum
        cols[rv[i]] = rv[i+1]
        i += 2
      end
      return cols
    end
