    def searchget(names = nil)
      raise ArgumentError if names && !names.is_a?(Array)
      args = Array.new(@args)
      if names
        args.push("get\0" + names.join("\0"))
      else
        args.push("get")
      end
      rv = @rdb.misc("search", args, RDB::MONOULOG)
      return Array.new if !rv
      for i in 0...rv.size
        cols = Hash.new
        cary = rv[i].split("\0")
        cnum = cary.size - 1
        j = 0
        while j < cnum
          cols[cary[j]] = cary[j+1]
          j += 2
        end
        rv[i] = cols
      end
      return rv
    end
